    label = 1 if label_str.upper().startswith("POS") else 0

# Weighted Ensemble Function
# =============================
def weighted_ensemble(preds, confidences=None, weights=None):
    if confidences is None:
        confidences = {k:1.0 for k in preds.keys()}
    if weights is None:
        weights = {k:1.0 for k in preds.keys()}
    
    vote_scores = {}
    for model_name, pred in preds.items():
        vote_scores[pred] = vote_scores.get(pred, 0) + weights.get(model_name,1.0) * confidences.get(model_name,1.0)
    
    final_label = max(vote_scores, key=vote_scores.get)
    return final_label

@app.route("/predict", methods=["POST"])
def predict_route():
    if "user" not in session:
        return jsonify({"error": "User not logged in."}), 401
    
    news_text = request.form.get("news_text", "").strip()
    if not news_text:
        return jsonify({"error": "Please provide news text."}), 400

    # Classical ML predictions
    X = vectorizer.transform([news_text])
    preds = {
        "LogisticRegression": int(lr_model.predict(X)[0]),
        "DecisionTree": int(dt_model.predict(X)[0]),
        "RandomForest": int(rf_model.predict(X)[0]),
        "GradientBoosting": int(gb_model.predict(X)[0])
    }

    # BERT prediction
    bert_pred, bert_score = predict_with_bert(news_text)
    preds["BERT"] = bert_pred
    confidences = {**{k:1.0 for k in preds.keys() if k!="BERT"}, "BERT": bert_score}

    # Model weights
    weights = {"LogisticRegression": 3, "DecisionTree": 1,
               "RandomForest": 2, "GradientBoosting": 2, "BERT": 5}

    # Weighted ensemble
    final_result = weighted_ensemble(preds, confidences, weights)

    # BERT high-confidence override
    BERT_CONF_THRESHOLD = 0.95
    if bert_score >= BERT_CONF_THRESHOLD:
        final_result = bert_pred

    # SerpAPI soft verification
    verified = verify_with_serpapi(news_text[:100])
    SERPAPI_OVERRIDE = True
    if SERPAPI_OVERRIDE and verified and final_result == 0:
        final_result = 1

    # Only return final result
    return jsonify({
        "FinalResult": "Real" if final_result == 1 else "Fake"
    })
